Solar System Animation

import numpy as np from js import document, requestAnimationFrame from solar_system import step, masses, positions, velocities, AU, SUN_IDX canvas = document.getElementById("sky") ctx = canvas.getContext("2d") scale = 150 / AU pos = positions.copy() vel = velocities.copy() dt = 60 * 60 # one hour # Colour per body: Sun, Mercury, Venus, Earth, Mars colors = ["#ffff00", "#bebebe", "#e5c39b", "#00a2ff", "#ff4500"] orbit_radii = np.linalg.norm(positions - positions[SUN_IDX], axis=1) def draw(*args): global pos, vel pos, vel = step(pos, vel, masses, dt) rel = pos - pos[SUN_IDX] ctx.clearRect(0, 0, canvas.width, canvas.height) # Draw orbital paths for i, r in enumerate(orbit_radii): if i == SUN_IDX: continue ctx.beginPath() ctx.arc(canvas.width/2, canvas.height/2, r*scale, 0, 2*np.pi) ctx.strokeStyle = "#333" ctx.stroke() # Draw bodies for i, (x, y, z) in enumerate(rel): px = canvas.width / 2 + x * scale py = canvas.height / 2 + y * scale ctx.beginPath() ctx.arc(px, py, 8 if i == SUN_IDX else 4, 0, 2 * np.pi) ctx.fillStyle = colors[i] ctx.fill() requestAnimationFrame(draw) requestAnimationFrame(draw)
